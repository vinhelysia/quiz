\section{CHƯƠNG 1: GIỚI THIỆU VỀ KINH TẾ HỌC}
% =========================================================================

\subsection{Hệ thống lý thuyết từ Slide}
\begin{itemize}
    \item \textbf{Sự khan hiếm:} Mong muốn vô hạn của con người luôn vượt quá nguồn lực hữu hạn của xã hội.
    \item \textbf{Chi phí cơ hội (Opportunity Cost):} Phần chi phí cao nhất bị mất đi do quyết định chọn một hoạt động thay thế khác. 
    \\ \textit{Phép thành ngữ dân gian}: ``Ăn đám giỗ, lỗ buổi cày''.
    \item \textbf{Phát biểu Thực chứng (Positive) và Chuẩn tắc (Normative):}
    \begin{itemize}
        \item \textbf{Thực chứng}: Mô tả và phân tích thế giới khách quan khách quan dựa trên số liệu. 
        \\ \textit{Ví dụ}: ``Lạm phát tăng khi chính phủ in quá nhiều tiền.''
        \item \textbf{Chuẩn tắc}: Khuyến nghị hoặc đưa ra lời khuyên mang tính chủ quan. 
        \\ \textit{Ví dụ}: ``Chính phủ cần giảm tốc độ lạm phát.''
    \end{itemize}
\end{itemize}

\subsection{Bài tập mẫu từ Slide}

\paragraph{Ví dụ (Độ dốc PPF và Chi phí cơ hội tăng dần):}
Cho bảng phối hợp sản lượng tối đa giữa 2 sản phẩm $X$ và $Y$ của một nền kinh tế:
\begin{center}
\begin{tabular}{cccccc}
\toprule
\textbf{Phối hợp} & \textbf{A} & \textbf{B} & \textbf{C} & \textbf{D} & \textbf{E} \\ \midrule
\textbf{X} & 0 & 50 & 100 & 150 & 200 \\
\textbf{Y} & 100 & 90 & 75 & 50 & 0 \\
\bottomrule
\end{tabular}
\end{center}
Hãy tính chi phí cơ hội biên để sản xuất thêm sản phẩm $X$ qua từng giai đoạn và nhận xét về hình dạng của đường PPF.

\paragraph{Lời giải:}
Chi phí cơ hội để sản xuất thêm $1$ đơn vị sản phẩm $X$ ($OC_X$) được tính theo công thức:
$$OC_X = \left| \frac{\Delta Y}{\Delta X} \right|$$
\begin{itemize}
    \item \textbf{Giai đoạn A $\rightarrow$ B:} $OC_X = \frac{100 - 90}{50 - 0} = \frac{10}{50} = 0.2$ đơn vị Y.
    \item \textbf{Giai đoạn B $\rightarrow$ C:} $OC_X = \frac{90 - 75}{100 - 50} = \frac{15}{50} = 0.3$ đơn vị Y.
    \item \textbf{Giai đoạn C $\rightarrow$ D:} $OC_X = \frac{75 - 50}{150 - 100} = \frac{25}{50} = 0.5$ đơn vị Y.
    \item \textbf{Giai đoạn D $\rightarrow$ E:} $OC_X = \frac{50 - 0}{200 - 150} = \frac{50}{50} = 1.0$ đơn vị Y.
\end{itemize}
\textbf{Nhận xét:} Chi phí cơ hội tăng dần ($0.2 \rightarrow 0.3 \rightarrow 0.5 \rightarrow 1.0$) cho thấy đường PPF cong \textbf{lồi hướng ra ngoài} gốc tọa độ.

\section{CHƯƠNG 2A: CẦU, CUNG VÀ CÂN BẰNG THỊ TRƯỜNG}
% =========================================================================

\subsection{Hệ thống công thức cốt lõi}
\begin{enumerate}
    \item \textbf{Phương trình cung cầu tuyến tính:}
    \begin{itemize}
        \item Hàm cầu: $Q_D = aP + b$ (với $a < 0$) hoặc dạng nghịch đảo $P = a'Q_D + b'$ (với $a' < 0$)
        \item Hàm cung: $Q_S = cP + d$ (với $c > 0$) hoặc dạng nghịch đảo $P = c'Q_S + d'$ (với $c' > 0$)
    \end{itemize}
    \item \textbf{Điểm cân bằng thị trường:}
    \begin{equation}
        Q_D = Q_S \quad \text{hoặc} \quad P_D(Q) = P_S(Q) \implies P_0, Q_0
    \end{equation}
    \item \textbf{Ba bước phân tích sự thay đổi trạng thái cân bằng:}
    \begin{itemize}
        \item \textbf{Bước 1}: Xác định sự kiện làm dịch chuyển đường cung, đường cầu hay cả hai.
        \item \textbf{Bước 2}: Xác định đường dịch chuyển sang trái hay sang phải.
        \item \textbf{Bước 3}: Sử dụng đồ thị cung cầu để so sánh giá và lượng cân bằng mới với ban đầu.
    \end{itemize}
\end{enumerate}

\subsection{Bài tập mẫu từ Slide}

\paragraph{Ví dụ 1 (Xác định cân bằng và tình trạng thị trường):}
Cho hàm số cung và cầu: $P = 2Q_S + 20$ và $P = -Q_D + 110$.
\begin{enumerate}
    \item Xác định giá và sản lượng cân bằng.
    \item Nếu giá thị trường hiện tại là $P = 50$, hãy xác định tình trạng thị trường (dư thừa hay thiếu hụt bao nhiêu)?
\end{enumerate}

\paragraph{Lời giải:}
\begin{enumerate}
    \item \textbf{Cân bằng thị trường:}
    Tại điểm cân bằng, ta có $P_D = P_S$:
    $$2Q + 20 = -Q + 110 \implies 3Q = 90 \implies Q_0 = 30$$
    Giá cân bằng: $P_0 = 2(30) + 20 = 80$.
    
    \item \textbf{Tại mức giá $P = 50$:}
    Thế giá $P = 50$ vào hàm cung và cầu nghịch đảo:
    \begin{itemize}
        \item Cung: $50 = 2Q_S + 20 \implies 2Q_S = 30 \implies Q_S = 15$.
        \item Cầu: $50 = -Q_D + 110 \implies Q_D = 60$.
    \end{itemize}
    So sánh: Vì $Q_D = 60 > Q_S = 15 \implies$ Thị trường đang xảy ra tình trạng \textbf{thiếu hụt} (lượng cầu vượt lượng cung) với lượng thiếu hụt là:
    $$\Delta Q = Q_D - Q_S = 60 - 15 = 45 \text{ đơn vị sản phẩm.}$$
\end{enumerate}

\paragraph{Ví dụ 2 (Viết hàm cung cầu từ số liệu):}
Cho bảng số liệu thị trường:
\begin{center}
\begin{tabular}{ccc}
\toprule
\textbf{P} & \textbf{$Q_D$} & \textbf{$Q_S$} \\ \midrule
10 & 200 & 40 \\
20 & 180 & 80 \\
30 & 160 & 120 \\
\bottomrule
\end{tabular}
\end{center}
Hãy viết phương trình cung cầu và tìm điểm cân bằng.

\paragraph{Lời giải:}
\begin{itemize}
    \item \textbf{Tìm hàm cầu:} Có dạng $Q_D = aP + b$. Thế số liệu vào hệ phương trình:
    \begin{equation*}
        \begin{cases}
            10a + b = 200 \\
            20a + b = 180
        \end{cases}
        \implies a = -2, b = 220 \implies Q_D = -2P + 220
    \end{equation*}
    \item \textbf{Tìm hàm cung:} Có dạng $Q_S = cP + d$. Thế số liệu vào hệ phương trình:
    \begin{equation*}
        \begin{cases}
            10c + d = 40 \\
            20c + d = 80
        \end{cases}
        \implies c = 4, d = 0 \implies Q_S = 4P
    \end{equation*}
    \item \textbf{Điểm cân bằng}: $Q_D = Q_S \implies -2P + 220 = 4P \implies 6P = 220 \implies P_0 \approx 36.67$, $Q_0 \approx 146.67$.
\end{itemize}

\section{CHƯƠNG 2B: ĐỘ CO GIÃN CỦA CẦU, CUNG VÀ ỨNG DỤNG}
% =========================================================================

\subsection{Hệ thống công thức cốt lõi}
\begin{enumerate}
    \item \textbf{Co giãn cầu theo giá ($E_D$):}
    \begin{itemize}
        \item \textit{Điểm}: $E_D = Q'_D(P) \times \frac{P}{Q}$
        \item \textit{Khoảng}: $E_D = \frac{Q_2 - Q_1}{P_2 - P_1} \times \frac{P_1 + P_2}{Q_1 + Q_2}$
    \end{itemize}
    \item \textbf{Quan hệ giữa co giãn và doanh thu ($TR = P \times Q$):}
    \begin{itemize}
        \item $|E_D| > 1$ (Co giãn nhiều): $P$ và $TR$ biến thiên ngược chiều $\implies$ Nên \textbf{giảm giá} để tăng doanh thu.
        \item $|E_D| < 1$ (Co giãn ít): $P$ và $TR$ biến thiên đồng chiều $\implies$ Nên \textbf{tăng giá} để tăng doanh thu.
    \end{itemize}
    \item \textbf{Co giãn chéo ($E_{XY}$):} $E_{XY} > 0$ (Thay thế), $E_{XY} < 0$ (Bổ sung).
    \item \textbf{Co giãn thu nhập ($E_I$):} $E_I < 0$ (Thứ cấp), $0 < E_I < 1$ (Thiết yếu), $E_I > 1$ (Xa xỉ).
\end{enumerate}

\subsection{Bài tập mẫu từ Slide}

\paragraph{Ví dụ 1 (Đánh giá co giãn chéo):}
\begin{itemize}
    \item \textbf{Tình huống 1}: Khi giá hàng hóa $X$ giảm $10\%$, lượng cầu hàng hóa $Y$ tăng $20\%$.
    \\ $\Rightarrow E_{XY} = \frac{20\%}{-10\%} = -2 < 0 \implies X$ và $Y$ là hai hàng hóa \textbf{bổ sung}.
    \item \textbf{Tình huống 2}: Giá bột giặt Omo tăng $15\%$ làm lượng cầu bột giặt Tide tăng $10\%$.
    \\ $\Rightarrow E_{XY} = \frac{10\%}{15\%} = 0.67 > 0 \implies$ Omo và Tide là hai hàng hóa \textbf{thay thế}.
\end{itemize}

\paragraph{Ví dụ 2 (Tính toán co giãn và tác động doanh thu):}
Cho hàm số: $Q_D = -0.5P + 200$ và $Q_S = 1.5P - 10$. Tính độ co giãn của cung và cầu tại điểm cân bằng ($P_0=105, Q_0=147.5$). Hãy cho biết để tăng doanh thu, nhà sản xuất nên tăng hay giảm giá?

\paragraph{Lời giải:}
\begin{itemize}
    \item Độ co giãn cầu theo giá tại cân bằng:
    $$E_D = Q'_D(P) \times \frac{P_0}{Q_0} = -0.5 \times \frac{105}{147.5} \approx -0.356$$
    \item Độ co giãn cung theo giá tại cân bằng:
    $$E_S = Q'_S(P) \times \frac{P_0}{Q_0} = 1.5 \times \frac{105}{147.5} \approx 1.068$$
    \item \textbf{Tăng doanh thu}: Vì $|E_D| \approx 0.356 < 1$ (cầu kém co giãn), doanh nghiệp muốn tăng doanh thu thì nên \textbf{tăng giá}.
\end{itemize}

\section{CHƯƠNG 2C: THẶNG DƯ VÀ CHÍNH SÁCH CỦA CHÍNH PHỦ}
% =========================================================================

\subsection{Hệ thống lý thuyết và công thức cốt lõi}
\begin{enumerate}
    \item \textbf{Thặng dư tiêu dùng (CS) và sản xuất (PS):}
    \begin{itemize}
        \item $CS$: phần diện tích dưới đường cầu, trên đường giá giao dịch.
        \item $PS$: phần diện tích trên đường cung, dưới đường giá giao dịch.
    \end{itemize}
    \item \textbf{Giá trần ($P_{\text{max}} < P_0$):} Gây ra thiếu hụt hàng hóa: $\Delta Q = Q_D - Q_S$.
    \item \textbf{Giá sàn ($P_{\text{min}} > P_0$):} Gây ra dư thừa hàng hóa: $\Delta Q = Q_S - Q_D$.
    \item \textbf{Chính sách Thuế ($t$):}
    \begin{equation}
        P_D - P_S = t \quad \text{và} \quad DWL = \frac{1}{2} \times t \times (Q_0 - Q_T)
    \end{equation}
    \item \textbf{Nguyên lý gánh nặng thuế:} Bên nào co giãn ít hơn sẽ chịu thuế nhiều hơn:
    $$\frac{t_{\text{tiêu dùng}}}{t_{\text{sản xuất}}} = \frac{E_S}{|E_D|}$$
\end{enumerate}

\subsection{Bài tập mẫu từ Slide}

\paragraph{Ví dụ 1 (Tổn thất vô ích do đánh thuế):}
Cho thị trường có hàm cung cầu: $P = -1.5Q_D + 160$ và $P = 0.5Q_S + 20$.
Chính phủ đánh thuế $t = 10$/sản phẩm lên người bán.
\begin{enumerate}
    \item Xác định giá và lượng cân bằng ban đầu.
    \item Xác định giá người mua trả ($P_D$), người bán nhận ($P_S$) và sản lượng mới ($Q_T$).
    \item Tính tổn thất vô ích ($DWL$).
\end{enumerate}

\paragraph{Lời giải:}
\begin{enumerate}
    \item \textbf{Cân bằng ban đầu:}
    $$-1.5Q + 160 = 0.5Q + 20 \implies 2Q = 140 \implies Q_0 = 70 \implies P_0 = 55$$
    \item \textbf{Khi có thuế $t = 10$:}
    $$P_D = P_S + 10 \implies -1.5Q_T + 160 = (0.5Q_T + 20) + 10$$
    $$\implies 2Q_T = 130 \implies Q_T = 65$$
    \begin{itemize}
        \item Giá người mua trả: $P_D = -1.5(65) + 160 = 62.5$.
        \item Giá người bán nhận: $P_S = 62.5 - 10 = 52.5$.
    \end{itemize}
    \item \textbf{Tổn thất vô ích (DWL):}
    $$DWL = \frac{1}{2} \times t \times (Q_0 - Q_T) = \frac{1}{2} \times 10 \times (70 - 65) = 25.$$
\end{enumerate}

\section{CHƯƠNG 3: LÝ THUYẾT HÀNH VI NGƯỜI TIÊU DÙNG}
% =========================================================================

\subsection{Hệ thống lý thuyết và công thức cốt lõi}
\begin{enumerate}
    \item \textbf{Đạo hàm tính hữu dụng biên:} $MU_X = \frac{\partial U}{\partial X}$, $MU_Y = \frac{\partial U}{\partial Y}$.
    \item \textbf{Hữu dụng biên rời rạc:} $MU = \frac{\Delta U}{\Delta X}$.
    \item \textbf{Tối ưu rổ hàng tiêu dùng:}
    \begin{equation}
        \begin{cases}
            \frac{MU_X}{P_X} = \frac{MU_Y}{P_Y} \quad \text{hay} \quad MRS_{XY} = \frac{P_X}{P_Y} \\
            X \cdot P_X + Y \cdot P_Y = I
        \end{cases}
    \end{equation}
\end{enumerate}

\subsection{Bài tập mẫu từ Slide}

\paragraph{Ví dụ 1 (Xác định MU từ U rời rạc):}
Cho tổng hữu dụng của $y$ như sau. Hãy tính hữu dụng biên của $y$ ($MU_y$):
\begin{center}
\begin{tabular}{cccccc}
\toprule
\textbf{$y$} & 0 & 2 & 5 & 8 & 10 \\ \midrule
\textbf{$U_y$} & 0 & 20 & 44 & 62 & 70 \\ \bottomrule
\end{tabular}
\end{center}
\paragraph{Lời giải:}
Áp dụng công thức hữu dụng biên rời rạc: $MU_y = \frac{U_{y2} - U_{y1}}{y_2 - y_1}$
\begin{itemize}
    \item Từ $y = 0 \to 2$: $MU_y = \frac{20 - 0}{2 - 0} = 10$.
    \item Từ $y = 2 \to 5$: $MU_y = \frac{44 - 20}{5 - 2} = 8$.
    \item Từ $y = 5 \to 8$: $MU_y = \frac{62 - 44}{8 - 5} = 6$.
    \item Từ $y = 8 \to 10$: $MU_y = \frac{70 - 62}{10 - 8} = 4$.
\end{itemize}

\paragraph{Ví dụ 2 (Bài tập rổ hàng tối ưu hệ số):}
Một người tiêu dùng có tổng hữu dụng $TU = (4X-8)Y$. Thu nhập $I = 30$, giá $P_X = 3, P_Y = 6$. Tìm lượng $X$ và $Y$ tối ưu.
\paragraph{Lời giải:}
\begin{enumerate}
    \item Đạo hàm riêng tính hữu dụng biên:
    $$MU_X = \frac{\partial TU}{\partial X} = 4Y, \quad MU_Y = \frac{\partial TU}{\partial Y} = 4X - 8$$
    \item Điều kiện tối ưu:
    $$\frac{MU_X}{P_X} = \frac{MU_Y}{P_Y} \implies \frac{4Y}{3} = \frac{4X-8}{6} \implies 8Y = 4X - 8 \implies X = 2Y + 2$$
    \item Điều kiện ngân sách:
    $$3X + 6Y = 30 \implies X + 2Y = 10$$
    \item Thế (1) vào (2):
    $$(2Y + 2) + 2Y = 10 \implies 4Y = 8 \implies Y^* = 2 \implies X^* = 6.$$
\end{enumerate}

\section{CHƯƠNG 4: LÝ THUYẾT SẢN XUẤT VÀ CHI PHÍ}
% =========================================================================

\subsection{Hệ thống công thức và Đạo hàm}
\begin{enumerate}
    \item \textbf{Năng suất biên và trung bình ngắn hạn:}
    $$AP_L = \frac{Q}{L}, \quad MP_L = \frac{dQ}{dL} = Q'(L)$$
    \item \textbf{Chi phí ngắn hạn ngắn liên tục:}
    \begin{equation}
        MC = \frac{dTC}{dQ} = (TC)' \quad \text{và} \quad MC = AVC \implies AVC_{\text{min}}
    \end{equation}
    \item \textbf{Quan hệ biên - trung bình qua Phép ẩn dụ tuổi trung bình của lớp học:}
    Thêm phần tử mới có giá trị cận biên lớn hơn trung bình hiện tại sẽ làm trung bình tăng lên, và ngược lại.
\end{enumerate}

\subsection{Bài tập mẫu từ Slide}

\paragraph{Ví dụ 1 (Tính toán bảng chi phí ngắn hạn):}
Cho số liệu chi phí ngắn hạn điền vào ô trống (Ví dụ 6 bài giảng):
\begin{center}
\begin{tabular}{cccccccc}
\toprule
\textbf{Q} & \textbf{TC} & \textbf{TFC} & \textbf{TVC} & \textbf{AC (ATC)} & \textbf{AVC} & \textbf{AFC} & \textbf{MC} \\ \midrule
\textbf{0} & 100 & 100 & 0 & - & - & - & - \\
\textbf{10} & 200 & 100 & 100 & 20 & 10 & 10 & 10 \\
\textbf{20} & 280 & 100 & 180 & 14 & 9 & 5 & 8 \\
\textbf{30} & 450 & 100 & 350 & 15 & 11.6 & 3.4 & 17 \\
\textbf{40} & 660 & 100 & 560 & 16.5 & 14 & 2.5 & 21 \\
\textbf{50} & ? & 100 & ? & ? & ? & ? & 25 \\
\bottomrule
\end{tabular}
\end{center}
Hãy điền đầy đủ dòng cuối cùng tại $Q = 50$.

\paragraph{Lời giải:}
\begin{itemize}
    \item Tính $TC_{50}$: $\Delta TC = MC \times \Delta Q \implies TC_{50} = TC_{40} + 25 \times (50 - 40) = 660 + 250 = 910$.
    \item $TVC_{50} = TC_{50} - TFC = 910 - 100 = 810$.
    \item $AC_{50} = 910 / 50 = 18.2$.
    \item $AVC_{50} = 810 / 50 = 16.2$.
    \item $AFC_{50} = 100 / 50 = 2.0$.
\end{itemize}

\paragraph{Ví dụ 2 (Hàm số chi phí liên tục):}
Cho hàm tổng chi phí: $TC = 2Q^3 - 20Q^2 + 100Q + 2500$. Xác định các hàm chi phí còn lại.
\paragraph{Lời giải:}
\begin{itemize}
    \item $TFC = 2500$, $TVC = 2Q^3 - 20Q^2 + 100Q$.
    \item $AVC = \frac{TVC}{Q} = 2Q^2 - 20Q + 100$.
    \item $ATC = \frac{TC}{Q} = 2Q^2 - 20Q + 100 + \frac{2500}{Q}$.
    \item $MC = (TC)' = 6Q^2 - 40Q + 100$.
\end{itemize}
