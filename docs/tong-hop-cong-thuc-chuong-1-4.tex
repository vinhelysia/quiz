\section{CHƯƠNG 1: GIỚI THIỆU VỀ KINH TẾ HỌC VI MÔ}
% =========================================================================

\subsection{Hệ thống công thức cốt lõi}
\begin{enumerate}
    \item \textbf{Chi phí cơ hội (Opportunity Cost - OC):}
    Chi phí cơ hội của việc chọn sản xuất thêm $1$ đơn vị sản phẩm $X$ là số lượng sản phẩm $Y$ phải từ bỏ:
    \begin{equation}
        OC_X = \left| \frac{\Delta Y}{\Delta X} \right| = \left| \frac{Y_2 - Y_1}{X_2 - X_1} \right|
    \end{equation}
    Nếu hàm PPF là đường cong liên tục biểu diễn bằng phương trình khả vi $Y = f(X)$, chi phí cơ hội biên tại một điểm chính là đạo hàm của hàm số (trị tuyệt đối độ dốc của tiếp tuyến tại điểm đó):
    \begin{equation}
        OC_X = \left| \frac{dY}{dX} \right| = |f'(X)|
    \end{equation}
    
    \item \textbf{Mối quan hệ chi phí cơ hội giữa hai sản phẩm:}
    \begin{equation}
        OC_Y = \frac{1}{OC_X}
    \end{equation}
\end{enumerate}

\subsection{Dạng hình học của đường PPF và ý nghĩa kinh tế}
\begin{itemize}
    \item \textbf{Đường PPF là đường thẳng:} Chi phí cơ hội là \textbf{hằng số} (không đổi). Các nguồn lực có thể thay thế hoàn hảo cho nhau trong việc sản xuất hai hàng hóa.
    \item \textbf{Đường PPF cong lồi so với gốc tọa độ (bowed-outward):} Chi phí cơ hội có xu hướng \textbf{tăng dần} khi ta sản xuất thêm nhiều sản phẩm $X$. Lý do: Các nguồn lực kinh tế không thích hợp như nhau để sản xuất cả hai sản phẩm.
\end{itemize}

\subsection{Nhận biết trạng thái các điểm sản xuất trên đồ thị PPF}
Cho một phương trình PPF $g(X, Y) = C$ (ví dụ: $X^2 + 2Y^2 = 100$):
\begin{itemize}
    \item \textbf{Điểm hiệu quả (Efficient):} Nằm trên đường PPF. Thế tọa độ $(X_0, Y_0)$ vào thỏa mãn đúng phương trình: $g(X_0, Y_0) = C$.
    \item \textbf{Điểm kém hiệu quả (Inefficient):} Nằm trong đường PPF (chưa tận dụng hết nguồn lực hoặc lãng phí). Thế tọa độ thỏa mãn: $g(X_0, Y_0) < C$.
    \item \textbf{Điểm bất khả thi (Unattainable):} Nằm ngoài đường PPF (không đủ nguồn lực để đạt tới). Thế tọa độ thỏa mãn: $g(X_0, Y_0) > C$.
\end{itemize}

\subsection{Quy trình giải bài tập mẫu}
\paragraph{Ví dụ minh họa:}
Cho phương trình giới hạn khả năng sản xuất là $2X^2 + 8Y = 200$. 
\begin{enumerate}
    \item Tìm chi phí cơ hội biên của sản phẩm $X$ tại phối hợp sản lượng $X = 6$.
    \item Xác định trạng thái của điểm $M(8, 10)$.
\end{enumerate}
\paragraph{Lời giải:}
\begin{enumerate}
    \item \textbf{Tính chi phí cơ hội biên của X:}
    Từ phương trình ta biến đổi $Y$ theo $X$: 
    \begin{equation*}
        8Y = 200 - 2X^2 \implies Y = 25 - 0.25X^2
    \end{equation*}
    Đạo hàm của $Y$ theo $X$:
    \begin{equation*}
        Y' = \frac{dY}{dX} = -0.5X
    \end{equation*}
    Tại $X = 6$, chi phí cơ hội để sản xuất thêm 1 đơn vị $X$ là:
    \begin{equation*}
        OC_X = |Y'(6)| = |-0.5 \times 6| = 3 \text{ đơn vị sản phẩm } Y.
    \end{equation*}
    
    \item \textbf{Xác định trạng thái của điểm M(8, 10):}
    Thế tọa độ điểm $M(8, 10)$ vào vế trái của phương trình PPF ban đầu:
    \begin{equation*}
        2(8^2) + 8(10) = 2(64) + 80 = 128 + 80 = 208
    \end{equation*}
    So sánh với vế phải: $208 > 200$.
    \\ $\Rightarrow$ Phối hợp sản lượng này nằm bên ngoài đường PPF $\Rightarrow$ \textbf{Bất khả thi} với trình độ công nghệ và nguồn lực hiện tại.
\end{enumerate}

\section{CHƯƠNG 2: CUNG - CẦU VÀ THỊ TRƯỜNG}
% =========================================================================

\subsection{Hệ thống công thức co giãn}
\begin{enumerate}
    \item \textbf{Độ co giãn của cầu theo giá ($E_D$):}
    \begin{itemize}
        \item \textit{Độ co giãn điểm:} 
        \begin{equation}
            E_D = \frac{dQ_D}{dP} \times \frac{P}{Q} = Q'_D(P) \times \frac{P}{Q}
        \end{equation}
        Lưu ý: Nếu đề bài cho hàm giá ngược $P = f(Q)$, ta có $Q'_D(P) = \frac{1}{P'_D(Q)} = \frac{1}{f'(Q)}$. Do đó:
        \begin{equation}
            E_D = \frac{1}{f'(Q)} \times \frac{P}{Q}
        \end{equation}
        \item \textit{Độ co giãn khoảng (trung điểm):}
        \begin{equation}
            E_D = \frac{\Delta Q}{\Delta P} \times \frac{\bar{P}}{\bar{Q}} = \frac{Q_2 - Q_1}{P_2 - P_1} \times \frac{(P_1 + P_2)/2}{(Q_1 + Q_2)/2}
        \end{equation}
    \end{itemize}

    \item \textbf{Mối quan hệ giữa $E_D$ và doanh thu doanh nghiệp (TR):}
    Ta có $TR = P \times Q$. Đạo hàm của $TR$ theo sản lượng $Q$ (Doanh thu biên - MR):
    \begin{equation}
        MR = \frac{dTR}{dQ} = P \left( 1 - \frac{1}{|E_D|} \right)
    \end{equation}
    \textbf{Nhận biết hành vi tăng doanh thu:}
    \begin{itemize}
        \item $|E_D| > 1$ (Cầu co giãn nhiều): $MR > 0 \implies Q \uparrow$ sẽ làm $TR \uparrow$. Để tăng doanh thu, doanh nghiệp nên \textbf{giảm giá} ($P \downarrow \implies Q \uparrow \uparrow \implies TR \uparrow$).
        \item $|E_D| < 1$ (Cầu co giãn ít): $MR < 0 \implies Q \uparrow$ sẽ làm $TR \downarrow$. Để tăng doanh thu, doanh nghiệp nên \textbf{tăng giá} ($P \uparrow \implies Q \downarrow \text{ ít} \implies TR \uparrow$).
        \item $|E_D| = 1$ (Cầu co giãn đơn vị): $MR = 0$. Doanh thu đạt giá trị \textbf{cực đại}.
    \end{itemize}

    \item \textbf{Độ co giãn chéo của cầu ($E_{XY}$):}
    \begin{equation}
        E_{XY} = Q'_{D_X}(P_Y) \times \frac{P_Y}{Q_X}
    \end{equation}
    \textbf{Dấu hiệu nhận biết mối quan hệ:}
    \begin{itemize}
        \item $E_{XY} > 0$: X và Y là hai hàng hóa \textbf{thay thế} (giá Y tăng làm người ta bỏ Y sang mua X).
        \item $E_{XY} < 0$: X và Y là hai hàng hóa \textbf{bổ sung} (giá Y tăng làm người ta ít mua Y và cũng ít mua X đi kèm).
        \item $E_{XY} = 0$: X và Y là hai hàng hóa \textbf{độc lập}.
    \end{itemize}

    \item \textbf{Độ co giãn của cầu theo thu nhập ($E_I$):}
    \begin{equation}
        E_I = Q'_D(I) \times \frac{I}{Q}
    \end{equation}
    \textbf{Dấu hiệu nhận biết tính chất hàng hóa:}
    \begin{itemize}
        \item $E_I < 0$: Hàng hóa \textbf{thứ cấp} (khi giàu lên người ta sẽ bớt mua hàng này).
        \item $E_I > 0$: Hàng hóa \textbf{thông thường}.
        \begin{itemize}
            \item $0 < E_I < 1$: Hàng hóa \textbf{thiết yếu} (nhu cầu tăng chậm hơn tốc độ tăng thu nhập).
            \item $E_I > 1$: Hàng hóa \textbf{xa xỉ} (nhu cầu tăng nhanh hơn tốc độ tăng thu nhập).
        \end{itemize}
    \end{itemize}
\end{enumerate}

\subsection{Thặng dư kinh tế và Tổn thất vô ích (DWL)}
\begin{enumerate}
    \item \textbf{Khi thị trường cân bằng tự do ($P_0, Q_0$):}
    \begin{itemize}
        \item Thặng dư tiêu dùng $CS$ nằm dưới đường cầu và trên đường giá cân bằng.
        \item Thặng dư sản xuất $PS$ nằm trên đường cung và dưới đường giá cân bằng.
    \end{itemize}
    
    \item \textbf{Khi đánh thuế ($t$ trên mỗi đơn vị):}
    Thị trường giao dịch tại sản lượng $Q_T < Q_0$.
    \begin{itemize}
        \item Giá người mua trả: $P_D$. Giá người bán nhận: $P_S = P_D - t$.
        \item Tổng doanh thu thuế của Chính phủ: $T = t \times Q_T$.
        \item Tổn thất vô ích của xã hội ($DWL$): diện tích tam giác bị mất đi do giảm sản lượng:
        \begin{equation}
            DWL = \frac{1}{2} \times t \times (Q_0 - Q_T)
        \end{equation}
    \end{itemize}
    
    \item \textbf{Sự phân chia gánh nặng thuế:}
    Tỷ lệ gánh nặng thuế giữa người tiêu dùng và nhà sản xuất tỷ lệ nghịch với độ co giãn của họ:
    \begin{equation}
        \frac{t_{\text{tiêu dùng}}}{t_{\text{sản xuất}}} = \frac{P_D - P_0}{P_0 - P_S} = \frac{E_S}{|E_D|}
    \end{equation}
    \begin{itemize}
        \item Bên nào \textbf{ít co giãn hơn} (ít nhạy cảm với giá hơn) sẽ gánh thuế \textbf{nhiều hơn}.
    \end{itemize}
\end{enumerate}

\subsection{Quy trình giải bài tập mẫu}
\paragraph{Ví dụ:}
Cho hàm cầu $Q_D = 100 - P$ và hàm cung $Q_S = P - 20$.
\begin{enumerate}
    \item Xác định giá và lượng cân bằng tự do.
    \item Nếu Chính phủ đánh thuế $t = 10$/sản phẩm lên người bán, tính giá người mua thực trả ($P_D$), giá người bán thực nhận ($P_S$) và tổn thất vô ích ($DWL$).
\end{enumerate}
\paragraph{Lời giải:}
\begin{enumerate}
    \item \textbf{Cân bằng tự do:}
    \begin{equation*}
        Q_D = Q_S \implies 100 - P = P - 20 \implies 2P = 120 \implies P_0 = 60
    \end{equation*}
    Thế vào hàm cầu: $Q_0 = 100 - 60 = 40$.
    
    \item \textbf{Có thuế $t = 10$:}
    Do đánh thuế vào người bán, khoảng cách giữa giá người mua trả và giá người bán nhận là $t = 10$.
    Ta có hệ phương trình:
    \begin{equation*}
        \begin{cases}
            P_D - P_S = 10 \implies P_D = P_S + 10\\
            Q_D(P_D) = Q_S(P_S) \implies 100 - P_D = P_S - 20
        \end{cases}
    \end{equation*}
    Thay thế $P_D$ ở phương trình (1) vào phương trình (2):
    \begin{equation*}
        100 - (P_S + 10) = P_S - 20 \implies 90 - P_S = P_S - 20 \implies 2P_S = 110 \implies P_S = 55
    \end{equation*}
    Giá người mua thực trả:
    \begin{equation*}
        P_D = 55 + 10 = 65
    \end{equation*}
    Lượng giao dịch mới:
    \begin{equation*}
        Q_T = 100 - 65 = 35
    \end{equation*}
    
    \item \textbf{Tính tổn thất vô ích (DWL):}
    Sử dụng công thức tính diện tích tổn thất vô ích:
    \begin{equation*}
        DWL = \frac{1}{2} \times t \times (Q_0 - Q_T) = \frac{1}{2} \times 10 \times (40 - 35) = 5 \times 5 = 25.
    \end{equation*}
\end{enumerate}

\section{CHƯƠNG 3: LÝ THUYẾT HÀNH VI NGƯỜI TIÊU DÙNG}
% =========================================================================

\subsection{Hệ thống công thức và Đạo hàm riêng}
\begin{enumerate}
    \item \textbf{Hữu dụng biên (Marginal Utility - MU):}
    Hữu dụng biên là đạo hàm riêng bậc một của hàm tổng thỏa dụng $U(X, Y)$:
    \begin{equation}
        MU_X = \frac{\partial U}{\partial X} = U'_X, \quad MU_Y = \frac{\partial U}{\partial Y} = U'_Y
    \end{equation}
    
    \item \textbf{Tỷ lệ thay thế biên (Marginal Rate of Substitution - MRS):}
    \begin{equation}
        MRS_{XY} = \frac{MU_X}{MU_Y}
    \end{equation}
    
    \item \textbf{Đường ngân sách:}
    \begin{equation}
        X \cdot P_X + Y \cdot P_Y = I
    \end{equation}
\end{enumerate}

\subsection{Phân loại các hàm thỏa dụng và Công thức giải nhanh}
Để giải trắc nghiệm siêu tốc, hãy nhận diện dạng hàm thỏa dụng ngay lập tức:

\begin{enumerate}[label=\alph*)]
    \item \textbf{Hàm Cobb-Douglas:} $U(X, Y) = X^\alpha Y^\beta$ (ví dụ: $U = X Y$ hoặc $U = X^2 Y^3$)
    \begin{itemize}
        \item Đây là dạng hàm có đường đẳng ích trơn, lồi về gốc tọa độ.
        \item \textbf{Công thức tối ưu giải nhanh:} Người tiêu dùng sẽ phân bổ thu nhập theo tỷ lệ số mũ:
        \begin{equation}
            X^* = \frac{\alpha}{\alpha + \beta} \times \frac{I}{P_X}, \quad Y^* = \frac{\beta}{\alpha + \beta} \times \frac{I}{P_Y}
        \end{equation}
    \end{itemize}

    \item \textbf{Hàng hóa thay thế hoàn hảo:} $U(X, Y) = aX + bY$ (ví dụ: $U = 2X + 3Y$)
    \begin{itemize}
        \item Đường đẳng ích là những đường thẳng. Tỷ lệ thay thế biên là hằng số: $MRS_{XY} = \frac{a}{b}$.
        \item So sánh $MRS_{XY}$ với tỷ giá thị trường $\frac{P_X}{P_Y}$ để tìm điểm tối ưu góc:
        \begin{itemize}
            \item Nếu $\frac{a}{b} > \frac{P_X}{P_Y}$: Người tiêu dùng chỉ mua $X \implies X^* = \frac{I}{P_X}, Y^* = 0$.
            \item Nếu $\frac{a}{b} < \frac{P_X}{P_Y}$: Người tiêu dùng chỉ mua $Y \implies X^* = 0, Y^* = \frac{I}{P_Y}$.
            \item Nếu $\frac{a}{b} = \frac{P_X}{P_Y}$: Bất kỳ phối hợp nào thỏa mãn đường ngân sách đều tối ưu.
        \end{itemize}
    \end{itemize}

    \item \textbf{Hàng hóa bổ sung hoàn hảo (Leontief):} $U(X, Y) = \min(aX, bY)$ (ví dụ: $U = \min(X, 2Y)$)
    \begin{itemize}
        \item Đường đẳng ích có dạng chữ L. Điểm tối ưu luôn nằm tại góc nhọn của chữ L, tức là:
        \begin{equation}
            aX = bY
        \end{equation}
        \item \textbf{Quy trình giải:} Giải hệ phương trình gồm $aX = bY$ và phương trình ngân sách $X \cdot P_X + Y \cdot P_Y = I$.
    \end{itemize}
\end{enumerate}

\subsection{Quy trình giải bài tập mẫu}
\paragraph{Ví dụ:}
Cho một người tiêu dùng có thu nhập $I = 240$ dùng để mua hai hàng hóa $X$ và $Y$ với giá tương ứng $P_X = 4$ và $P_Y = 3$. Hãy xác định rổ hàng tối ưu cho các trường hợp hàm thỏa dụng sau:
\begin{enumerate}
    \item $U(X, Y) = X^2 Y$
    \item $U(X, Y) = 2X + Y$
    \item $U(X, Y) = \min(2X, Y)$
\end{enumerate}
\paragraph{Lời giải:}
\begin{enumerate}
    \item \textbf{Trường hợp hàm Cobb-Douglas $U = X^2 Y^1$ (với $\alpha = 2, \beta = 1$):}
    Áp dụng công thức giải nhanh:
    \begin{equation*}
        X^* = \frac{2}{2 + 1} \times \frac{240}{4} = \frac{2}{3} \times 60 = 40
    \end{equation*}
    \begin{equation*}
        Y^* = \frac{1}{2 + 1} \times \frac{240}{3} = \frac{1}{3} \times 80 = 26.67
    \end{equation*}
    
    \item \textbf{Trường hợp thay thế hoàn hảo $U = 2X + Y$ (với $a = 2, b = 1$):}
    Tính tỷ lệ thay thế biên: $MRS_{XY} = \frac{a}{b} = \frac{2}{1} = 2$.
    Tính tỷ giá thị trường: $\frac{P_X}{P_Y} = \frac{4}{3} \approx 1.33$.
    So sánh: Vì $MRS_{XY} = 2 > 1.33 = \frac{P_X}{P_Y}$, hàng hóa $X$ mang lại độ thỏa dụng biên trên mỗi đồng chi tiêu lớn hơn $Y$.
    Do đó, người tiêu dùng chỉ mua $X$:
    \begin{equation*}
        X^* = \frac{I}{P_X} = \frac{240}{4} = 60, \quad Y^* = 0.
    \end{equation*}

    \item \textbf{Trường hợp bổ sung hoàn hảo $U = \min(2X, Y)$:}
    Thiết lập điều kiện tại góc chữ L: $2X = Y \implies Y = 2X$.
    Thế vào phương trình ngân sách:
    \begin{equation*}
        X \cdot P_X + Y \cdot P_Y = I \implies 4X + 3(2X) = 240 \implies 10X = 240 \implies X^* = 24
    \end{equation*}
    Tìm $Y$: $Y^* = 2(24) = 48$.
\end{enumerate}

\section{CHƯƠNG 4: LÝ THUYẾT SẢN XUẤT VÀ CHI PHÍ}
% =========================================================================

\subsection{Lý thuyết sản xuất ngắn hạn và dài hạn}
\begin{enumerate}
    \item \textbf{Năng suất biên của Lao động ($MP_L$) và Năng suất trung bình ($AP_L$):}
    \begin{equation}
        MP_L = \frac{dQ}{dL} = Q'(L), \quad AP_L = \frac{Q}{L}
    \end{equation}
    \textbf{Nhận biết mối quan hệ:}
    \begin{itemize}
        \item $MP_L > AP_L \implies AP_L$ đang tăng.
        \item $MP_L < AP_L \implies AP_L$ đang giảm.
        \item $MP_L = AP_L \implies AP_L$ đạt giá trị \textbf{cực đại} (đây là điểm giao nhau của đường $MP_L$ và $AP_L$).
    \end{itemize}

    \item \textbf{Hiệu suất theo quy mô của hàm Cobb-Douglas $Q = A \cdot L^\alpha K^\beta$:}
    \begin{itemize}
        \item $\alpha + \beta > 1$: Hiệu suất \textbf{tăng} theo quy mô (Q tăng nhiều hơn tỷ lệ tăng đầu vào).
        \item $\alpha + \beta = 1$: Hiệu suất \textbf{không đổi} theo quy mô.
        \item $\alpha + \beta < 1$: Hiệu suất \textbf{giảm} theo quy mô.
    \end{itemize}

    \item \textbf{Điều kiện tối ưu phối hợp đầu vào (đạt đường mở rộng sản xuất):}
    \begin{equation}
        \frac{MP_L}{P_L} = \frac{MP_K}{P_K} \quad \text{hay} \quad \frac{MP_L}{MP_K} = \frac{P_L}{P_K}
    \end{equation}
\end{enumerate}

\subsection{Cấu trúc chi phí ngắn hạn và phép toán vi phân}
Đố với các hàm số liên tục, chi phí biên là đạo hàm riêng của tổng chi phí $TC$ (hoặc chi phí biến đổi $TVC$) theo sản lượng $Q$:
\begin{equation}
    MC = \frac{dTC}{dQ} = (TC)' \quad \text{hoặc} \quad MC = \frac{dTVC}{dQ} = (TVC)'
\end{equation}
Ngược lại, nếu biết hàm chi phí biên $MC$ và chi phí cố định $TFC$, ta tìm lại hàm tổng chi phí $TC$ bằng tích phân:
\begin{equation}
    TC = \int MC \, dQ + TFC
\end{equation}

\textbf{Quan hệ giữa các hàm chi phí trung bình và chi phí biên:}
\begin{itemize}
    \item $MC < ATC \implies ATC$ giảm dọc theo sản lượng.
    \item $MC > ATC \implies ATC$ tăng dọc theo sản lượng.
    \item $MC = ATC \implies ATC$ đạt cực tiểu. Ta tìm điểm cực tiểu của $ATC$ bằng cách giải phương trình:
    \begin{equation}
        MC = ATC \quad \text{hoặc} \quad (ATC)' = 0
    \end{equation}
    \item Tương tự, đối với chi phí biến đổi trung bình $AVC$:
    \begin{equation}
        MC = AVC \quad \text{hoặc} \quad (AVC)' = 0 \implies AVC \text{ cực tiểu}
    \end{equation}
\end{itemize}

\textbf{Mối liên hệ giữa chi phí và năng suất:}
\begin{equation}
    MC = \frac{P_L}{MP_L}, \quad AVC = \frac{P_L}{AP_L}
\end{equation}

\subsection{Quy trình giải bài tập mẫu}
\paragraph{Ví dụ:}
Cho hàm tổng chi phí ngắn hạn của một doanh nghiệp là $TC = Q^3 - 10Q^2 + 37Q + 128$.
\begin{enumerate}
    \item Tìm hàm chi phí biên $MC$, chi phí biến đổi trung bình $AVC$.
    \item Xác định mức sản lượng $Q$ để chi phí biến đổi trung bình đạt cực tiểu và tính giá trị cực tiểu đó.
    \item Tìm lại hàm $TC$ biết $MC = 3Q^2 - 20Q + 37$ và chi phí cố định ban đầu là $128$.
\end{enumerate}
\paragraph{Lời giải:}
\begin{enumerate}
    \item \textbf{Tìm các hàm chi phí:}
    \begin{itemize}
        \item Chi phí biên $MC$ là đạo hàm của $TC$:
        \begin{equation*}
            MC = (TC)' = \frac{d}{dQ}(Q^3 - 10Q^2 + 37Q + 128) = 3Q^2 - 20Q + 37
        \end{equation*}
        \item Chi phí cố định: $TFC = 128$ (không phụ thuộc vào $Q$).
        \item Chi phí biến đổi: $TVC = TC - TFC = Q^3 - 10Q^2 + 37Q$.
        \item Chi phí biến đổi trung bình $AVC$:
        \begin{equation*}
            AVC = \frac{TVC}{Q} = Q^2 - 10Q + 37
        \end{equation*}
    \end{itemize}
    
    \item \textbf{Cực tiểu hóa AVC:}
    \textbf{Cách 1: Cho đạo hàm của AVC bằng 0}
    \begin{equation*}
        (AVC)' = 2Q - 10 = 0 \implies Q = 5
    \end{equation*}
    \textbf{Cách 2: Giải phương trình $MC = AVC$}
    \begin{align*}
        3Q^2 - 20Q + 37 = Q^2 - 10Q + 37 &\implies 2Q^2 - 10Q = 0 \\
        &\implies 2Q(Q - 5) = 0 \implies Q = 5 \quad (\text{loại } Q = 0)
    \end{align*}
    Tại $Q = 5$, giá trị cực tiểu của $AVC$ là:
    \begin{equation*}
        AVC_{\text{min}} = 5^2 - 10(5) + 37 = 25 - 50 + 37 = 12.
    \end{equation*}

    \item \textbf{Khôi phục hàm $TC$ từ $MC$ và $TFC$:}
    \begin{align*}
        TC &= \int MC \, dQ = \int (3Q^2 - 20Q + 37) dQ \\
        &= Q^3 - 10Q^2 + 37Q + C
    \end{align*}
    Vì khi $Q = 0$ ta có $TC = TFC = 128 \implies C = 128$.
    Vậy hàm tổng chi phí là $TC = Q^3 - 10Q^2 + 37Q + 128$.
\end{enumerate}
